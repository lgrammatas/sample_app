#### Git

# Setup
git config --global user.name "name"  
git config --global user.email l.grammatas@gmail.com
.gitignore # files to ignore
git init # initialize git files
git remote add origin git@github.com:lgrammatas/first_app.git
git remote set-url origin <SSH-URL> # to enable SSH


#Branches
git checkout -b modify-README # create a new branch
git checkout -f # reset the branch
git commit -am "Stupid change" # Commit a change
git branch # see all the branches
git branch -D <branch2delete> # delete
git merge <branch>

#Standard
git status # get the status
git log # get the changes that have been made
git add . # add all the files
git commit -m "Initial commit" # Commits the changes in the buffer
hit push
